    var use_analytic = false
    var f_val: Vector2
    var f_prime: Vector2

    # Check if we can use analytic derivatives
    if Config.input_function_type == Config.ComplexFunc.IDENTITY:
        if Config.function_type == Config.ComplexFunc.ZETA:

            f_val = res[0]
            f_prime = res[1]
            use_analytic = true
        elif Config.function_type == Config.ComplexFunc.DIRICHLET_ETA:
            var res = dirichlet_eta_with_derivatives(z.x, z.y, Config.iterations)
            f_val = res[0]
            f_prime = res[1]
            use_analytic = true

    if not use_analytic:
        var p_ref = Config.complex_to_world(z.x, z.y)
        f_val = get_field(p_ref.x, p_ref.y)
        var delta_x = 1e-5
        var p_ref_dx = Config.complex_to_world(z.x + delta_x, z.y)
        var f_val_dx = get_field(p_ref_dx.x, p_ref_dx.y)
        f_prime = (f_val_dx - f_val) / delta_x

Is the finite difference `f_prime` also with respect to `z`?
Let's trace the finite difference:
```gdscript
	var p_ref = Config.complex_to_world(z.x, z.y)
	var f_val = get_field(p_ref.x, p_ref.y)
	var delta_x = 1e-5
	var p_ref_dx = Config.complex_to_world(z.x + delta_x, z.y)

```
Wait! The finite difference adds `delta_x` to `z.x`, which is the real part of the complex coordinate. Then it converts to world, gets field, and computes `(f_val_dx - f_val) / delta_x`.
Because `get_field` takes world coordinates but eventually calls `get_field_at` on the complex coordinate, `f_val` is simply `F(z)`.
So `(f_val_dx - f_val) / delta_x` is indeed the complex derivative $F'(z)$, but assuming $F$ is holomorphic so we can just use the partial derivative with respect to $x$.
This matches the exact output of `zeta_with_derivatives(z.x, z.y).dx`.
